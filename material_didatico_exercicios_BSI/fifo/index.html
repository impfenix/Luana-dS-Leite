// FilaCircular.cpp
#include <iostream>

class FilaCircular {
private:
    int *dados;
    int inicio, fim, qt, MAX;

public:
    FilaCircular(int tamanho) { /* ... */ }
    ~FilaCircular() { /* ... */ }

    bool estaVazia() {
        return qt == 0;
    }
    bool estaCheia() {
        return qt == MAX;
    }

    // Função para inserir um elemento na fila
    void inserir(int valor) {
        // Passo 1: Verificação
        if (estaCheia()) { return; }

        // Passo 2: Inserir o dado
        dados[fim] = valor;

        // Passo 3: Atualizar ponteiros
        fim = (fim + 1) % MAX;
        qt++;
    }

    // Função para remover um elemento da fila
    void remover() {
        // Passo 1: Verificação
        if (estaVazia()) { return; }

        // Passo 2: Remover o dado (simulado)
        // Em C++, apenas moveríamos o ponteiro 'inicio'.
        // O valor antigo permanece, mas é inacessível.

        // Passo 3: Atualizar ponteiros
        inicio = (inicio + 1) % MAX;
        qt--;
    }
};
