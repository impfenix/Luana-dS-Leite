// FilaCircular.cpp
#include <iostream>

class FilaCircular {
private:
    int *dados;
    int inicio, fim, qt, MAX;

public:
    FilaCircular(int tamanho) {
        MAX = tamanho;
        dados = new int[MAX];
        inicio = 0;
        fim = 0;
        qt = 0;
    }

    ~FilaCircular() {
        delete[] dados;
    }

    bool estaVazia() { return qt == 0; }
    bool estaCheia() { return qt == MAX; }

    // Função para inserir um elemento na fila
    void inserir(int valor) {
        if (estaCheia()) {
            std::cout << "ERRO: Fila cheia!\n";
            return;
        }
        dados[fim] = valor;
        fim = (fim + 1) % MAX;
        qt++;
    }

    // Função para remover um elemento da fila
    void remover() {
        if (estaVazia()) {
            std::cout << "ERRO: Fila vazia!\n";
            return;
        }
        // Em uma implementação real, retornaríamos dados[inicio]
        inicio = (inicio + 1) % MAX;
        qt--;
    }
};
